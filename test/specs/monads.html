o 'of(a, f) <=> f.of(a)' do for-all(Any).satisfy (a) -> _.of(a)(new Id(a)).is-equal new Id(a) and \ _.of(a)(SId).is-equal new SId(a) .as-test! o 'ap(a, b) <=> a.ap(b)' do for-all(Any).satisfy (a) -> _.ap(new Id(-> [it, it]))(new Id(a)).is-equal new Id([a, a]) .as-test!
