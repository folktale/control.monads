Function: concat

Concatenates two semigroups.

  • type: (Semigroup s) => s(a) -> s(a) -> s(a)
export concat = (a, b) --> a ++ b

Function: empty

Constructs a new empty semigroup. + type: (Semigroup s) => s -> s(a)

export empty = (a) -> | a.empty => a.empty! | otherwise => a@@empty!

Function: map

Maps over a Functor instance. + type: (Functor f) => (a -> b) -> f(a) -> f(b)

export map = (f, a) --> a.map f

Function: of_

Constructs a new applicative instance. + type: (Applicative f) => a -> f(a)

export of_ = (a, f) --> | f.of => f.of a | _ => f@@of a module.exports.of = of_

Function: ap

Applies the function of an Applicative to the value of another Applicative. + type: (Applicative f) => f(a -> b) -> f(a) -> f(b)

export ap = (a, b) --> a.ap b

Function: chain

Extracts the value of a monad.

  • type: (Chain c) => (a -> c(b)) -> c(a) -> c(b)
export chain = (f, a) --> a.chain f
